2、修改自己的对应的SQL语句
使用方式'#(对象.属性)'或'#(属性)'

使用条件为 “=”时

**固定写法：where 数据库字段名 =  COALESCE( NULLIF('#(对象.属性)', ''), 数据库字段 )**
参考以下示例倒数第二、三行

```sql
WITH RECURSIVE org_tree AS (
    SELECT 
        o.id,
        o.parent_org_id,
        o.org_name,
        o.org_code,
        o.org_level_code,
        o.org_type,
        o.biz_type,
        o.node_type,
				o.tenant_id
    FROM yee_sys_org o
    WHERE o.org_code = '7611282360'
    UNION ALL
    SELECT 
        c.id,
        c.parent_org_id,
        c.org_name,
        c.org_code,
        c.org_level_code,
        c.org_type,
        c.biz_type,
        c.node_type,
				c.tenant_id
    FROM yee_sys_org c
    INNER JOIN org_tree t ON c.parent_org_id = t.id
    WHERE c.org_type IN (
        1,2,3,
        613643492274606080,
        613643493176381440,
        613643495806210048
    )
),
main_data AS (
    SELECT
        ot.id AS org_id,   
        x.xfmc,
        ROUND(x.zyywsrze,2) AS zyywsrze,
        ROUND(x.cbfyze,2)  AS cbfyze,
        (x.cbfyze / x.zyywsrze) AS ratio,
				ot.tenant_id
    FROM xjydxypt_xfszc_tzhjcsj x
     
    INNER JOIN org_tree ot ON x.oa_dept_code = ot.org_code
    WHERE x.month = '2025-11'
)
SELECT
    ROW_NUMBER() OVER (ORDER BY ratio DESC) AS sort_no,
    org_id,   
    concat(ratio,'#sort#',xfmc) xfmc,
    zyywsrze,
    cbfyze,
    ratio,
		tenant_id
FROM main_data 
where tenant_id =  COALESCE( NULLIF('#(user.tenantId)', ''), tenant_id )  
ORDER BY ratio DESC limit 10
```

使用条件为 “in”时


**固定写法：where (数据库字段 IN ('#(对象.属性)') OR concat('#(对象.属性)') IS NULL)  **
参考以下示例倒数第二、三行

```sql
WITH RECURSIVE org_tree AS (
    SELECT 
        o.id,
        o.parent_org_id,
        o.org_name,
        o.org_code,
        o.org_level_code,
        o.org_type,
        o.biz_type,
        o.node_type,
				o.tenant_id
    FROM yee_sys_org o
    WHERE o.org_code = '7611282360'

    UNION ALL

    SELECT 
        c.id,
        c.parent_org_id,
        c.org_name,
        c.org_code,
        c.org_level_code,
        c.org_type,
        c.biz_type,
        c.node_type,
				c.tenant_id
    FROM yee_sys_org c
    INNER JOIN org_tree t ON c.parent_org_id = t.id
    WHERE c.org_type IN (
        1,2,3,
        613643492274606080,
        613643493176381440,
        613643495806210048
    )
),
main_data AS (
    SELECT
        ot.id AS org_id,   
        x.xfmc,
        ROUND(x.zyywsrze,2) AS zyywsrze,
        ROUND(x.cbfyze,2)  AS cbfyze,
        (x.cbfyze / x.zyywsrze) AS ratio,
				ot.tenant_id
    FROM xjydxypt_xfszc_tzhjcsj x
     
    INNER JOIN org_tree ot ON x.oa_dept_code = ot.org_code
    WHERE x.month = '2025-11'
)

SELECT
    ROW_NUMBER() OVER (ORDER BY ratio DESC) AS sort_no,
    org_id,   
    concat(ratio,'#sort#',xfmc) xfmc,
    zyywsrze,
    cbfyze,
    ratio,
		tenant_id
FROM main_data 
where (tenant_id IN ('#(user.ids)') OR concat('#(user.ids)') IS NULL)  
ORDER BY ratio DESC limit 10
```
